/// Usage ///



// so to start, include BracketsGF/brackets.hpp:


#pragma comment(lib, "user32.lib")
#pragma comment(lib, "gdi32.lib")
#pragma comment(lib, "kernel32.lib")

#include "BracketsGF/brackets.hpp"


// then you initialize a WinMain function:


int WINAPI WinMain(
    HINSTANCE handle_to_instance, 
    HINSTANCE handle_to_previous, 
    LPSTR long_pointer_to_command, 
    int show_window_command
) { // ...


// then you initialize your window and the following parameters:


WINDOW window = {};
window.window_title = L"window title";
window.window_class = L"window class";
window.rectangles = std::make_unique<std::vector<RECTANGLE>>(); // required

window.window_procedure_data.previous_window_width = 800; // required base width
window.window_procedure_data.previous_window_height = 600; // required base height


// then you initialize your window class:


WNDCLASSW window_class = {};
window_class.lpfnWndProc = window_procedure; // window_procedure is a function
window_class.hInstance = handle_to_instance;
window_class.lpszClassName = window.window_class;
window_class.hCursor = LoadCursorW(NULL, (LPCWSTR)IDC_ARROW);
RegisterClassW(&window_class);


// then you need to initialize your window.handle_to_window:


window.handle_to_window = CreateWindowExW(
    0, window.window_class, window.window_title,
    WS_OVERLAPPEDWINDOW | WS_VISIBLE, 
    CW_USEDEFAULT, CW_USEDEFAULT, 800, 600, 
    nullptr, nullptr, handle_to_instance, &window
);


// then you can initialize your rectangles like this:


RECTANGLE green_square = {};
green_square.base_x = 100.0;
green_square.base_y = 100.0;
green_square.base_width = 150;
green_square.base_height = 150;

green_square.x = green_square.base_x;
green_square.y = green_square.base_y;
green_square.width = green_square.base_width;
green_square.height = green_square.base_height;

green_square.color = RGB(0, 255, 0);

update_rectangle(green_square); // sync
add_rectangle(&green_square, window);


// then finally run the message loop:


    return run_message_loop();
}
